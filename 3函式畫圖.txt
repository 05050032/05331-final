function s=t(first,last,dx,func1,func2,func3)
  msg=nargchk(0,6,nargin);
  error(msg);
  x=first:dx:last;
  y=feval(func1,x);
  plot(x,y,'b-');
  hold on;
  y=feval(func2,x);
  plot(x,y,'k--');
  hold on;
  y=feval(func3,x);
  plot(x,y,'r:');
endfunction
